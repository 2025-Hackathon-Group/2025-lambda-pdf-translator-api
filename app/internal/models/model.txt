Table Organization {
  id int [pk]
  name String
  email String [unique]
}

Table User {
  id int [pk]
  email String [unique]
  name String
  password String
  profilePicture String [unique]
  createdAt datetime
  updatedAt datetime
  organizationID int
}

Table BaseDocument {
  id int [pk]
  documentTitle String [unique]
  docType enum
  uploadedAt datetime
  lastUsed datetime
  belongsToUserID int
}

Table TranslatedDocument {
  id int [pk]
  translatedTitle String [unique]
  translatedFrom String
  translatedTo String
  translatedAt datetime
  baseDocumentID int
}

Table Chat {
  id int [pk]
  lastChat datetime
  firstCreated datetime
  baseDocumentID int
}

Table Messages{
  id int [pk]
  sender enum
  content String
  sentAt datetime
  chatID int
}

Ref: User.organizationID > Organization.id
Ref: TranslatedDocument.baseDocumentID > BaseDocument.id
Ref: Messages.chatID > Chat.id
Ref: Chat.baseDocumentID > BaseDocument.id
Ref: BaseDocument.belongsToUserID > User.id
